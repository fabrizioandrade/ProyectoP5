/**

 *
 * @function
 * @memberof module:app
 * @name morgan
 * @param {string} format - El formato de registro que se utilizar치.

 * Middleware para analizar los datos JSON de la solicitud.

 * @function

    resave: false,
    saveUninitialized: false,

 * Middleware para analizar las cookies en las solicitudes.
